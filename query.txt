1. Search top 100 artists with the most events

SELECT COUNT(*) event_num, e.name FROM Events e 
JOIN Artist a ON a.id = e.artist_id
GROUP BY e.name
ORDER BY e.event_num DESC

2. Upcoming events in 30 days

SELECT name FROM Event 
ORDER BY Time DESC
